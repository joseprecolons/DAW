SELECT PELLICULES.titol_peli, PRESTECS.num_exemplar, SOCIS.nom_soci, PRESTECS.data_prestec, PRESTECS.data_retorn, DATEDIFF(PRESTECS.data_retorn, PRESTECS.data_prestec) Dies, DATEDIFF(PRESTECS.data_retorn, PRESTECS.data_prestec) * 2.5 Lloguer FROM PELLICULES, PRESTECS, SOCIS WHERE PELLICULES.id_peli = PRESTECS.id_peli AND PRESTECS.id_soci = SOCIS.id_soci AND DATEDIFF(PRESTECS.data_retorn, PRESTECS.data_prestec) > 3 ORDER BY DATEDIFF(PRESTECS.data_retorn, PRESTECS.data_prestec) ASC, PELLICULES.titol_peli, PRESTECS.num_exemplar;
